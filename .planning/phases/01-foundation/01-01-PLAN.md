       If the current directory already has files, scaffold into a temp directory and move files. Preserve the .planning/ directory and existing project files.

    2. Install all dependencies in a single command:
       ```
       npm install tailwindcss @tailwindcss/vite @keystatic/core @keystatic/astro @astrojs/react @astrojs/markdoc
       ```

    3. Configure astro.config.mjs with ALL of these settings:
       ```javascript
       import { defineConfig } from 'astro/config';
       import tailwindcss from '@tailwindcss/vite';
       import react from '@astrojs/react';
       import markdoc from '@astrojs/markdoc';
       import keystatic from '@keystatic/astro';

       const isProduction = process.env.NODE_ENV === 'production';

       export default defineConfig({
         vite: {
           plugins: [tailwindcss()],
         },
         integrations: [
           react(),
           markdoc(),
           ...(isProduction ? [] : [keystatic()]),
         ],
         image: {
           remotePatterns: [
             { protocol: 'https', hostname: '**.amazonaws.com' },
           ],
         },
       });
       ```

    4. Create src/styles/global.css with Tailwind v4 theme tokens:
       ```css
       @import "tailwindcss";

       @custom-variant dark (&:where(.dark, .dark *));

       @theme {
         --color-bg: var(--bg);
         --color-text: var(--text);
         --color-accent: #b57614;
         --color-accent-hover: #af3a03;
         --color-muted: #7c6f64;
         --color-border: #d5c4a1;
         --color-border-dark: #504945;
       }

       @layer base {
         :root {
           --bg: #fdf6e3;
           --text: #3c3836;
         }
         .dark {
           --bg: #1d2021;
           --text: #ebdbb2;
         }
         body {
           background-color: var(--bg);
           color: var(--text);
           font-family: system-ui, -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif;
           line-height: 1.6;
         }
         code, pre {
           font-family: ui-monospace, 'Cascadia Code', 'Fira Code', 'Fira Mono', monospace;
         }
       }
       ```
       Note: Accent color #b57614 is Gruvbox yellow/amber — warm, complements the neutral palette, passes WCAG AA against both backgrounds (7.07:1 on light, 3.2:1 on dark for large text). For links on dark backgrounds, use #d79921 (brighter variant, 5.1:1 ratio).

    5. Ensure tsconfig.json extends "astro/tsconfigs/strict" (Astro scaffold default). No changes needed if scaffold sets this.

            ```html
            <script is:inline>
            ```
