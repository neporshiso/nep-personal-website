3. Create `src/pages/rss.xml.ts` with RSS 2.0 format (simpler, broader reader support than Atom):

```typescript
import rss from '@astrojs/rss';
import { getCollection } from 'astro:content';
import type { APIContext } from 'astro';

export async function GET(context: APIContext) {
  const allPosts = await getCollection('posts');
  const posts = allPosts
    .filter((p) => !p.data.draft)
    .sort((a, b) => b.data.publishedDate.valueOf() - a.data.publishedDate.valueOf());

  return rss({
    title: "nep's thoughts",
    description: "Blog posts by Nep — reflections on software development, learning, and building things on the web.",
    site: context.site!,
    items: posts.map((post) => ({
      title: post.data.title,
      pubDate: post.data.publishedDate,
      description: post.data.excerpt || '',
      link: `/thoughts/${post.id}/`,
    })),
  });
}
```

Use excerpt-only in feed items (not full content) — keeps the feed lightweight and encourages clicking through to the site.

IMPORTANT: The link must use `/thoughts/${post.id}/` — NOT `/blog/`. The route was renamed to /thoughts during Phase 2 smoke test.
